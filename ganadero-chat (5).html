function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Registros'); if (!sheet) { sheet = ss.insertSheet('Registros'); crearEncabezados(sheet); } if (sheet.getLastRow() === 0) { crearEncabezados(sheet); } sheet.appendRow([ data.fecha || Utilities.formatDate(new Date(), 'America/Argentina/Buenos_Aires', 'dd/MM/yyyy HH:mm'), data.tipo || '', data.cantidad || '', data.sexo || '', data.potrero || '', data.destino || '', data.vacuna || '', data.vacunado || '', data.peso_kg || '', data.precio_kg || '', data.observaciones || '', data.caravana || '' ]); var ultima = sheet.getLastRow(); var tipo = (data.tipo || '').toLowerCase(); var colores = { 'nacimiento':'#E8F5E9','vacunacion':'#F3E5F5','peso':'#E3F2FD', 'venta':'#FFF8E1','compra':'#E0F7FA','muerte':'#FFEBEE','traspaso':'#FFF3E0' }; if (colores[tipo]) sheet.getRange(ultima, 1, 1, 12).setBackground(colores[tipo]); sheet.getRange(ultima, 3).setHorizontalAlignment('center'); // Cantidad sheet.getRange(ultima, 4).setHorizontalAlignment('center'); // Sexo sheet.getRange(ultima, 8).setHorizontalAlignment('center'); // Vacunado return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function crearEncabezados(sheet) { var h = [ 'Fecha y Hora', 'Tipo', 'Cantidad', 'Sexo', 'Potrero Origen', 'Potrero Destino', 'Vacuna', 'Vacunado', 'Peso (kg)', 'Precio/kg', 'Observaciones', 'N Caravana' ]; sheet.appendRow(h); var r = sheet.getRange(1, 1, 1, h.length); r.setFontWeight('bold') .setBackground('#2D4020') .setFontColor('#FFFFFF') .setFontSize(10) .setHorizontalAlignment('center'); sheet.setFrozenRows(1); var anchos = [145, 110, 80, 85, 150, 150, 130, 85, 85, 90, 200, 120]; for (var i = 0; i < anchos.length; i++) sheet.setColumnWidth(i + 1, anchos[i]); try { sheet.getRange(1, 1, sheet.getMaxRows(), h.length).createFilter(); } catch(e) {} } function doGet(e) { return ContentService.createTextOutput('GanaderoBot activo'); } // Ejecutar manualmente UNA VEZ para reinicializar la planilla function inicializarPlanilla() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Registros'); if (sheet) { // Limpiar sin borrar la hoja sheet.clearContents(); sheet.clearFormats(); sheet.clearNotes(); // Eliminar filtro si existe try { if (sheet.getFilter()) sheet.getFilter().remove(); } catch(e) {} } else { sheet = ss.insertSheet('Registros'); } crearEncabezados(sheet); SpreadsheetApp.getUi().alert('Planilla reinicializada! 12 columnas limpias con filtros.'); }
